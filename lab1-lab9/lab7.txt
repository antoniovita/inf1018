1) Os dois programas fazem a mesma coisa: usam um ponteiro para percorrer o array S2 até achar o byte 0, o terminador NULL. Para cada byte, imprimem o caractere com printf("%c") e ao final imprimem “\n” e retornam 0.
No .data do assembly ficam os mesmos dados do C: S2 é o vetor de bytes, Sf é a string “%c” e Sf2 é a string “\n”.
No main, o prólogo só monta e desmonta o frame da função e salva registradores (rbx e r12). O retorno 0 é feito colocando 0 em rax.
O ponteiro pc do C é representado por r12: movq $S2, %r12 inicializa pc com o endereço de S2, cmpb $0,(%r12) e JE L2 implementam o teste do while, movsbl (%r12),%eax carrega *pc como int, a chamada printf passa os argumentos em rdi (formato) e rsi (caractere) e zera eax por ser função variádica, addq $1,%r12 faz pc++ e o loop repete. Após sair, chama printf("\n") usando Sf2.

2) Código modificado em C:
#include <stdio.h>

char S2[] = {65,108,111,32,123,103,97,108,101,114,97,125,33,0};

int main(void) {
    char *pc = S2;
    while (*pc) {
        if (*pc != 123 && *pc != 125) printf("%c", *pc);
        pc++;
    }
    printf("\n");
    return 0;
}

Código modificado em Assembly:

.data
S2:    .byte 65, 108, 111, 32, 123, 103, 97, 108, 101, 114, 97, 125, 33, 0
Sf:  .string "%c"    /* primeira string de formato para printf */
Sf2: .string "\n"    /* segunda string de formato para printf */

.text
.globl  main
main:

  pushq   %rbp
  movq    %rsp, %rbp
  subq    $16, %rsp
  movq    %rbx, -8(%rbp)  /* guarda rbx */
  movq    %r12, -16(%rbp)  /* guarda r12 */

  movq  $S2, %r12  /* r12 = &S2 */

L1:
  cmpb  $0, (%r12)  /* if (*pc == 0) ? */
  je  L3          /* goto L3 */
  
  cmpb $123, (%r12)
  je L2

  cmpb $125, (%r12)
  je L2

  movsbl  (%r12), %eax    /* eax = *r12 (estendendo o byte para 32 bits */

/*************************************************************/
/* este trecho imprime o valor de %eax (estraga %eax)  */
  movq    $Sf, %rdi    /* primeiro parametro (ponteiro)*/
  movl    %eax, %esi   /* segundo parametro  (inteiro) */
  movl  $0, %eax
  call  printf       /* chama a funcao da biblioteca */
/*************************************************************/

L2:
  addq  $1, %r12  /* r12 += 1; */
  jmp  L1         /* goto L1; */

L3:  
/*************************************************************/
/* este trecho imprime o \n (estraga %eax)                  */
  movq    $Sf2, %rdi    /* primeiro parametro (ponteiro)*/
  movl  $0, %eax
  call  printf       /* chama a funcao da biblioteca */
/*************************************************************/

/***************************************************************/
/* mantenha este trecho aqui e nao mexa - finalizacao!!!!      */
  movq  $0, %rax  /* rax = 0  (valor de retorno) */
  movq    -16(%rbp), %r12 /* recupera r12 */
  movq    -8(%rbp), %rbx  /* recupera rbx */
  leave
  ret      
/***************************************************************/




3) int quadradados(void) {

    int quadrado = i * i;
    printf("%d\n", quadrado);
  }
  return 0;
}

  jmp   L1

L2:
  movq  $0, %rax
  popq  %r12
  popq  %rbx
